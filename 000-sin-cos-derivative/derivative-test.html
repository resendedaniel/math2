plotChart <- function(f, n=1001) {
    curve(f, -tau, tau)
    
    breaks <- c(1,
                3/4,
                1/2,
                1/4,
                0)
    for(b in breaks) {
        abline(v=-tau*b)
        abline(v=tau*b)
    }
}
